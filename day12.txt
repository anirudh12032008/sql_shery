USE HR;
SELECT FIRST_NAME, LAST_NAME FROM EMPLOYEES WHERE DEPARTMENT_ID = (SELECT DEPARTMENT_ID FROM EMPLOYEES WHERE EMPLOYEE_ID = 110);
SELECT FIRST_NAME, CASE WHEN SALARY<(SELECT AVG(SALARY) FROM EMPLOYEES) THEN 'UNDERPAID' ELSE 'OVERPAID' END AS PAY_STATUS FROM EMPLOYEES;
